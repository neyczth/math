    \begin{kong}\rm
        {\kai (本题满分10分)}

        求所有的非零实数$k$, 使得关于$x$的方程$kx^2+(4k+1)x+3-k=0$的根都是整数.
    \end{kong}

    \vspace*{9cm}

    \begin{kong}\rm
        {\kai (本题满分10分)}

        证明:

        (1)若$x>0,y>0$, 则$\dfrac{x^2}{x+y}\geq\dfrac{3x-y}{4}$;

        (2)若$x,y,z>0$, 则$\dfrac{x^3}{x+y}+\dfrac{y^3}{y+z}+\dfrac{z^3}{z+x}\geq\dfrac{xy+yz+zx}{2}$.
    \end{kong}
